          function step5() {
            var count = $('.step5').length + 1;

            return `<div class='step step5'\>
              <p>
              ${content(count)}
              <p>
            <div>`;
          }


          function content(count) {
            if( count < 3 ) {
              return `There are ${count} step5 divs`;
            } else if (count < 5) {
              return 'Stop clicking step4 already!!';
            } else {
              return window.location = 'chapter3.html';
            }
          }
